建構SQL資料表:


CREATE TABLE studenttab(
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(20),
    chi INT,
    math INT,
    eng INT
);
    


顯示資料表的編碼狀態:

SHOW CREATE TABLE studenttab;



修改資料表的編碼型態:

ALTER TABLE studenttab
CONVERT To CHARACTER SET utf8mb4
COLLATE utf8mb4_unicode_ci;


修改資料庫的編碼:

ALTER DATABASE testuser
CONVERT To CHARACTER SET utf8mb4
COLLATE utf8mb4_unicode_ci;


插入資料

INSERT INTO studenttab(name,chi,eng,math) VALUES ('小琪',85,75,90)


查詢

SELECT * FROM studenttab;   (*指所有欄位)

SELECT name,math FROM studenttab WHERE math>=90;   (加入條件)

SELECT name,math FROM studenttab ORDER BY mat DESC;


更新成績

UPDATE studenttab SET eng=50 WHERE name='小琪';
